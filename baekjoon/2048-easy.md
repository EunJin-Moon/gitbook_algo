import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.ArrayList;
import java.util.List;
import java.util.StringTokenizer;

public class Main_12100_2048 {
	static int N;

	public static void main(String[] args) throws IOException {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		StringTokenizer st;
		N = Integer.parseInt(br.readLine());
		int[][] map = new int[N][N];

		for (int i = 0; i < N; i++) {
			st = new StringTokenizer(br.readLine());
			for (int j = 0; j < N; j++) {
				map[i][j] = Integer.parseInt(st.nextToken());
			}
		}

		// 다섯번 이동시켜 가장 큰 값 찾기
		ans = Integer.MIN_VALUE;

		solve(map, 0);
		
		
		System.out.println(ans);

	}

	static int ans;

	static void solve(int[][] map, int cnt) {
		if (cnt == 5) {
			findNum(map);
			return;
		}
		for (int k = 0; k < 4; k++) {
			int[][] arr = copy(map);
			move(k, arr);
			solve(arr, cnt + 1);
		}

	}

	static int[][] copy(int[][] map) {
		int[][] copy = new int[N][N];
		for (int i = 0; i < N; i++) {
			for (int j = 0; j < N; j++) {
				copy[i][j] = map[i][j];
			}
		}
		return copy;
	}

	static void move(int dir, int[][] map) {
		List<Integer> list;
		if (dir == 0) { // 위
			for (int j = 0; j < N; j++) {
				list = new ArrayList<>();
				for (int i = 0; i < N; i++) {
					if (map[i][j] != 0)
						list.add(map[i][j]);
					map[i][j] = 0;
				}
				list = calList(list);

				for (int k = 0; k < list.size(); k++) {
					map[k][j] = list.get(k);
				}
			}
		} else if (dir == 1) {// 아래
			for (int j = 0; j < N; j++) {
				list = new ArrayList<>();
				for (int i = N - 1; i >= 0; i--) {
					if (map[i][j] != 0)
						list.add(map[i][j]);
					map[i][j] = 0;
				}
				list = calList(list);
				for (int k = 0; k < list.size(); k++) {
					map[N - 1 - k][j] = list.get(k);
				}
			}

		} else if (dir == 2) { // 좌

			for (int i = 0; i < N; i++) {
				
				list = new ArrayList<>();
				for (int j = 0; j < N; j++) {
					if (map[i][j] != 0)
						list.add(map[i][j]);
					map[i][j] = 0;
				}
				list = calList(list);

				for (int k = 0; k < list.size(); k++) {
					map[i][k] = list.get(k);
				}

			}
		} else { // 우

			for (int i = 0; i < N; i++) {
				list = new ArrayList<>();
				for (int j = N - 1; j >= 0; j--) {
					if (map[i][j] != 0)
						list.add(map[i][j]);
					map[i][j] = 0;
				}
				list = calList(list);
				for (int k = 0; k < list.size(); k++) {
					map[i][N - 1 - k] = list.get(k);
				}
			}
		}
	}

	static List<Integer> calList(List<Integer> list) {
		for (int k = 0; k < list.size() - 1; k++) {
			if (list.get(k).equals( list.get(k + 1))) {
				list.set(k, list.get(k) * 2);
				list.remove(k + 1);
			}
		}
		return list;
	}

	static void findNum(int[][] map) {
		for (int i = 0; i < N; i++) {
			for (int j = 0; j < N; j++) {
				ans = Math.max(ans, map[i][j]);

			}
		}
	}

}
